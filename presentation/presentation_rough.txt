// Opening the process pipe
void LinuxProcessSource::open(...) {
    pipe = popen(commandToRun.c_str(), "r");
    if (!pipe) {
        throw InvalidConfigParameter("Could not start process");
    }
}

// Reading data into tuple buffer
FillTupleBufferResult LinuxProcessSource::fillTupleBuffer(...) {
    auto mem = tupleBuffer.getAvailableMemoryArea();
    const size_t bytesRead = std::fread(mem.data(), 1, mem.size(), pipe);
    
    if (std::feof(pipe)) return FillTupleBufferResult::eos();
    return FillTupleBufferResult::withBytes(bytesRead);
}
