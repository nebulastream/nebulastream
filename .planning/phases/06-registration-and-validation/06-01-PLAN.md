1. Create `nes-sources/rust/nes-source-runtime/src/file_sink.rs`:
   ```rust
   use std::path::PathBuf;
   use tokio::fs::File;
   use tokio::io::AsyncWriteExt;
   use crate::sink::AsyncSink;
   use crate::sink_context::{SinkContext, SinkMessage};
   use crate::sink_error::SinkError;

   pub struct AsyncFileSink {
       path: PathBuf,
   }

   impl AsyncFileSink {
       pub fn new(path: PathBuf) -> Self {
           Self { path }
       }
   }

   impl AsyncSink for AsyncFileSink {
       async fn run(&mut self, ctx: &SinkContext) -> Result<(), SinkError> {
           let mut file = File::create(&self.path).await
               .map_err(|e| SinkError::new(format!("failed to open file: {}", e)))?;
           loop {
               match ctx.recv().await {
                   SinkMessage::Data(buf) => {
                       file.write_all(buf.as_slice()).await
                           .map_err(|e| SinkError::new(format!("write failed: {}", e)))?;
                   }
                   SinkMessage::Flush => {
                       file.flush().await
                           .map_err(|e| SinkError::new(format!("flush failed: {}", e)))?;
                   }
                   SinkMessage::Close => {
                       file.flush().await
                           .map_err(|e| SinkError::new(format!("final flush failed: {}", e)))?;
                       return Ok(());
                   }
               }
           }
       }
   }
   ```

   fn spawn_file_sink(
       sink_id: u64,
       file_path: &str,
       channel_capacity: u32,
       error_fn_ptr: usize,
       error_ctx_ptr: usize,
   ) -> Result<Box<SinkHandle>, String> {
       let error_fn: nes_source_runtime::sink::ErrorFnPtr = unsafe {
           std::mem::transmute::<usize, nes_source_runtime::sink::ErrorFnPtr>(error_fn_ptr)
       };
       let error_ctx = error_ctx_ptr as *mut std::ffi::c_void;

       let sink = nes_source_runtime::AsyncFileSink::new(std::path::PathBuf::from(file_path));

           sink_id,
           sink,
           channel_capacity as usize,
           error_fn,
           error_ctx,
       );
       Ok(Box::new(SinkHandle { inner: *inner }))
   }
