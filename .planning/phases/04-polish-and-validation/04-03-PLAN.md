       export abstract class NesLexerBase extends Lexer {

           const nextChar = this.inputStream.LA(1);

    7. Create `src/lib/sql/validateSql.ts`:
       ```typescript
       import { CharStream, CommonTokenStream } from 'antlr4ng';
       import { AntlrSQLLexer } from './AntlrSQLLexer.js';
       import { AntlrSQLParser } from './AntlrSQLParser.js';

       export interface SqlError { line: number; column: number; message: string; }

       export function validateSql(sql: string): SqlError[] {
         const inputStream = CharStream.fromString(sql);
         const lexer = new AntlrSQLLexer(inputStream);
         const tokenStream = new CommonTokenStream(lexer);
         const parser = new AntlrSQLParser(tokenStream);
         const collector = { errors: [] as SqlError[], syntaxError(_r: unknown, _o: unknown, line: number, col: number, msg: string) { this.errors.push({ line, column: col, message: msg }); } };
         lexer.removeErrorListeners();
         parser.removeErrorListeners();
         lexer.addErrorListener(collector);
         parser.addErrorListener(collector);
         parser.singleStatement();
         return collector.errors;
       }
       ```
