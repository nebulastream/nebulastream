Add a test case with YAML containing a File source that has no `file_path` in its configuration:
```cpp
TEST_F(TopologyValidatorTest, FileSourceMissingFilePathReturnsError)
{
    // YAML with a File source missing the required file_path parameter
    const std::string yaml = R"(
workers:
  - host: worker1
    grpcPort: 4000
    dataPort: 5000
logicalSources:
  - name: test_source
    schema:
      - name: id
        type: UINT64
    physicalSources:
      - type: File
        config: {}
)";
    auto errors = NES::Validator::validateTopology(yaml);
    ASSERT_FALSE(errors.empty());
    // Check that at least one error mentions the missing config or file_path
    bool foundConfigError = false;
    for (const auto& err : errors)
    {
        if (err.find("file_path") != std::string::npos || err.find("required") != std::string::npos || err.find("missing") != std::string::npos || err.find("File") != std::string::npos)
        {
            foundConfigError = true;
            break;
        }
    }
    ASSERT_TRUE(foundConfigError) << "Expected error about missing file_path for File source";
}
```
