       ```
       std::string validateTopology(const std::string& yamlString) {
           try {
               // 1. Parse YAML
               auto config = YAML::Load(yamlString).as<TopologyConfig>();
               // 2. Populate source catalog
               auto sourceCatalog = std::make_shared<SourceCatalog>();
               for each logical source: sourceCatalog->addLogicalSource(name, schema)
               for each physical source: sourceCatalog->addPhysicalSource(...)
               // 3. Populate sink catalog
               auto sinkCatalog = std::make_shared<SinkCatalog>();
               for each sink: sinkCatalog->addSinkDescriptor(...)
               // 4. Validate queries via StatementBinder
               auto binder = StatementBinder{sourceCatalog, ...};
               for each query: binder.parseAndBind(query) — check result
               return ""; // success
           } catch (const std::exception& e) {
               return e.what();
           }
       }
       ```

       - Handle the case where LegacyOptimizer may be needed: start WITHOUT it (per research recommendation). If parseAndBind alone doesn't catch source/type errors, investigate adding it.
